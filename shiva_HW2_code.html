def invertedIndex(tokenizedDocList : list):

                List = invertedIndex[term] 
                List.append(str(i)) 
                invertedIndex[term] = List

                invertedIndex[term] = [str(i)]

def takeQuery(query): 
    
    #splits query into k-grams
    bigrams = createBigram(query) 
    print("The bigrams generated form wildcard query: " + str(bigrams) + "\n") 
    print("The words from k-gram indx are: ")

    sections = query.split("*")

    finalList = [] 
    for item in postFilterList:

    for word in postFilterList:

    return documentList
